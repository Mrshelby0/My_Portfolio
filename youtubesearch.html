TakeCommand Function

    def TakeCommand():
        r = sr.Recognizer()
        with sr.Microphone()as sources:
            print(":Listening............")
            r.pause_threshold =1
            audio = r.listion(sources)
    
        try:
            print(":Recognizing.........")
            query = r.recognize_google(audio,language='en-in')
            print(f" : your command:{query}\n")
        
        except:
            return ""
        return query.lower()
